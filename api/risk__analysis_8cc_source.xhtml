39 void RiskAnalysis::Analyze() noexcept {
40  // Set the seed for the pseudo-random number generator if given explicitly.
41  // Otherwise it defaults to the implementation dependent value.
42  if (Analysis::settings().seed() >= 0)
43  Random::seed(Analysis::settings().seed());
44 
45  for (const mef::FaultTreePtr& ft : model_->fault_trees()) {
46  for (const mef::Gate* target : ft->top_events()) {
47  LOG(INFO) << "Running analysis: " << target->id();
48  RunAnalysis(target->id(), *target);
49  LOG(INFO) << "Finished analysis: " << target->id();
50  }
51  }
52 }

54 void RiskAnalysis::RunAnalysis(const std::string& name,
55  const mef::Gate& target) noexcept {
56  const std::string& algorithm = Analysis::settings().algorithm();
57  if (algorithm == "bdd") {
58  RunAnalysis<Bdd>(name, target);
59  } else if (algorithm == "zbdd") {
60  RunAnalysis<Zbdd>(name, target);
61  } else {
62  assert(algorithm == "mocus");
63  RunAnalysis<Mocus>(name, target);
64  }
65 }

67 template <class Algorithm>
68 void RiskAnalysis::RunAnalysis(const std::string& name,
69  const mef::Gate& target) noexcept {
70  auto* fta = new FaultTreeAnalyzer<Algorithm>(target, Analysis::settings());
71  fta->Analyze();
72  if (Analysis::settings().probability_analysis()) {
73  if (Analysis::settings().approximation() == "no") {
74  RunAnalysis<Algorithm, Bdd>(name, fta);
75  } else if (Analysis::settings().approximation() == "rare-event") {
76  RunAnalysis<Algorithm, RareEventCalculator>(name, fta);
77  } else {
78  assert(Analysis::settings().approximation() == "mcub");
79  RunAnalysis<Algorithm, McubCalculator>(name, fta);
80  }
81  }
82  fault_tree_analyses_.emplace(name, ext::make_unique(fta));
83 }

85 template <class Algorithm, class Calculator>
86 void RiskAnalysis::RunAnalysis(const std::string& name,
87  FaultTreeAnalyzer<Algorithm>* fta) noexcept {
88  auto* pa = new ProbabilityAnalyzer<Calculator>(fta);
89  pa->Analyze();
90  if (Analysis::settings().importance_analysis()) {
91  auto* ia = new ImportanceAnalyzer<Calculator>(pa);
92  ia->Analyze();
93  importance_analyses_.emplace(name, ext::make_unique(ia));
94  }
95  if (Analysis::settings().uncertainty_analysis()) {
96  auto* ua = new UncertaintyAnalyzer<Calculator>(pa);
97  ua->Analyze();
98  uncertainty_analyses_.emplace(name, ext::make_unique(ua));
99  }
100  probability_analyses_.emplace(name, ext::make_unique(pa));
101 }
