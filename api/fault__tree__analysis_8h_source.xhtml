54 /// Collection of unique literals.
55 /// This collection is designed as a space-efficient drop-in replacement
56 /// for const std::vector<const Literal>.
57 ///
58 /// The major differences from std::vector:
59 ///
60 /// 1. Iterators are proxies.
61 /// 2. The size is fixed upon construction.
62 /// 3. The size is limited to 32 literals.
63 /// 4. Only necessary and required API is provided for the use-cases.
64 /// 5. The construction is handled differently.
65 class Product {
66  public:
67  /// Read iterator over Product Literals.
68  class iterator {
69  /// Equality test operators.
70  /// @{
71  friend bool operator==(const iterator& lhs, const iterator& rhs) {
72  return &lhs.product_ == &rhs.product_ && lhs.pos_ == rhs.pos_;
73  }
74  friend bool operator!=(const iterator& lhs, const iterator& rhs) {
75  return !(lhs == rhs);
76  }
77  /// @}
78 
79  public:
80  /// Constructs an iterator on a position in a specific product.
81  ///
82  /// @param[in] pos Valid position.
83  /// @param[in] product The host container.
84  iterator(int pos, const Product& product) : pos_(pos), product_(product) {}
85 
86  /// @returns The literal at the position.
87  Literal operator*() const {
88  return {product_.GetComplement(pos_), *product_.data_[pos_]};
89  }
90 
91  /// Pre-increment operator for for-range loops.
92  iterator& operator++() {
93  ++pos_;
94  return *this;
95  }
96 
97  private:
98  int pos_; ///< The current position of the iterator.
99  const Product& product_; ///< The container to be iterated over.
100  };

102  /// Initializes product literals.
103  ///
104  /// @tparam GeneratorIterator Iterator type with its operator* returning
105  /// std::pair or struct with
106  /// ``first`` being complement flag of the Literal,
107  /// ``second`` being pointer to the event.
108  ///
109  /// @param[in] size The number of literals in the product.
110  /// @param[in] it The generator of literals for this product.
111  template <class GeneratorIterator>
112  Product(int size, GeneratorIterator it) noexcept
113  : size_(size),
114  complement_vector_(0),
115  data_(new const mef::BasicEvent*[size]) {
116  assert(size >= 0 && size <= 32);
117  for (int i = 0; i < size; ++i, ++it) {
118  const auto& literal = *it;
119  if (literal.first)
120  complement_vector_ |= 1 << i; // The complement flag.
121  data_[i] = literal.second; // The variable itself.
122  }
123  }
124 
125  /// Move constructor for products to store in a database.
126  ///
127  /// @param[in] other Fully initialized product.
128  Product(Product&& other) noexcept
129  : size_(other.size_),
130  complement_vector_(other.complement_vector_),
131  data_(std::move(other.data_)) {
132  other.size_ = other.complement_vector_ = 0;
133  }
134 
135  /// @returns true for unity product with no literals.
136  bool empty() const { return size_ == 0; }
137 
138  /// @returns The number of literals in the product.
139  int size() const { return size_; }
140 
141  /// @returns A read proxy iterator that points to the first element.
142  iterator begin() const { return iterator(0, *this); }
143 
144  /// @returns A sentinel iterator signifying the end of iteration.
145  iterator end() const { return iterator(size_, *this); }
146 
147  private:
148  /// Determines if the literal is complement.
149  ///
150  /// @param[in] pos The position of the literal.
151  ///
152  /// @returns true if the literal is complement.
153  bool GetComplement(int pos) const {
154  return complement_vector_ & (1 << pos);
155  }
156 
157  std::uint8_t size_; ///< The number of literals in the product.
158  std::uint32_t complement_vector_; ///< The complement flags of the literals.
159  /// The collection of literal events.
160  std::unique_ptr<const mef::BasicEvent*[]> data_;
161 };

192 /// Fault tree analysis functionality.
193 /// The analysis must be done on
194 /// a validated and fully initialized fault trees.
195 /// After initialization of the analysis,
196 /// the fault tree under analysis should not change;
197 /// otherwise, the success of the analysis is not guaranteed,
198 /// and the results may be invalid.
199 /// After the requested analysis is done,
200 /// the fault tree can be changed without restrictions.
201 /// However, other analyses may rely on unchanged fault tree
202 /// to use the results of this fault tree analysis.
203 ///
204 /// To conduct a new analysis on the changed fault tree,
205 /// a new FaultTreeAnalysis object must be created.
206 /// In general, rerunning the same analysis twice
207 /// will mess up the analysis and corrupt the previous results.
208 ///
209 /// @warning Run analysis only once.
210 /// One analysis per FaultTreeAnalysis object.
211 class FaultTreeAnalysis : public Analysis {
212  public:
213  /// Traverses a valid fault tree from the root gate
214  /// to collect databases of events, gates,
215  /// and other members of the fault tree.
216  /// The passed fault tree must be pre-validated without cycles,
217  /// and its events must be fully initialized.
218  ///
219  /// @param[in] root The top event of the fault tree to analyze.
220  /// @param[in] settings Analysis settings for all calculations.
221  ///
222  /// @note It is assumed that analysis is done only once.
223  ///
224  /// @warning If the fault tree structure is changed,
225  /// this analysis does not incorporate the changed structure.
226  /// Moreover, the analysis results may get corrupted.
227  FaultTreeAnalysis(const mef::Gate& root, const Settings& settings);
228 
229  virtual ~FaultTreeAnalysis() = default;
230 
231  /// @returns The top gate that is passed to the analysis.
232  const mef::Gate& top_event() const { return top_event_; }
233 
234  /// Analyzes the fault tree and performs computations.
235  /// This function must be called
236  /// only after initializing the fault tree
237  /// with or without its probabilities.
238  ///
239  /// @note This function is expected to be called only once.
240  ///
241  /// @warning If the original fault tree is invalid,
242  /// this function will not throw or indicate any errors.
243  /// The behavior is undefined for invalid fault trees.
244  /// @warning If the fault tree structure has changed
245  /// since the construction of the analysis,
246  /// the analysis will be invalid or fail.
247  virtual void Analyze() noexcept = 0;
248 
249  /// @returns A set of Boolean products as the analysis results.
250  const std::vector<Product>& products() const { return products_; }
251 
252  /// @returns Collection of basic events that are in the products.
253  const std::unordered_set<const mef::BasicEvent*>& product_events() const {
254  return product_events_;
255  }
256 
257  protected:
258  /// Converts resultant sets of basic event indices to strings
259  /// for future reporting.
260  /// This function also collects basic events in products.
261  ///
262  /// @param[in] results A sets with indices of events from calculations.
263  /// @param[in] graph Boolean graph with basic event indices and pointers.
264  void Convert(const std::vector<std::vector<int>>& results,
265  const BooleanGraph* graph) noexcept;
266 
267  private:
268  const mef::Gate& top_event_; ///< The root of the graph under analysis.
269  std::vector<Product> products_; ///< Container of analysis results.
270  /// The set of events in the resultant products.
271  std::unordered_set<const mef::BasicEvent*> product_events_;
272 };

301 template <class Algorithm>
302 void FaultTreeAnalyzer<Algorithm>::Analyze() noexcept {
303  CLOCK(analysis_time);
304 
305  CLOCK(graph_creation);
306  graph_ = std::make_unique<BooleanGraph>(FaultTreeAnalysis::top_event(),
307  Analysis::settings().ccf_analysis());
308  LOG(DEBUG2) << "Boolean graph is created in " << DUR(graph_creation);
309 
310  CLOCK(prep_time); // Overall preprocessing time.
311  LOG(DEBUG2) << "Preprocessing...";
312  Preprocessor* preprocessor = new CustomPreprocessor<Algorithm>(graph_.get());
313  preprocessor->Run();
314  delete preprocessor; // No exceptions are expected.
315  LOG(DEBUG2) << "Finished preprocessing in " << DUR(prep_time);
316 #ifndef NDEBUG
317  if (Analysis::settings().preprocessor)
318  return; // Preprocessor only option.
319 #endif
320  CLOCK(algo_time);
321  LOG(DEBUG2) << "Launching the algorithm...";
322  algorithm_ = std::make_unique<Algorithm>(graph_.get(), Analysis::settings());
323  algorithm_->Analyze();
324  LOG(DEBUG2) << "The algorithm finished in " << DUR(algo_time);
325  LOG(DEBUG2) << "# of products: " << algorithm_->products().size();
326 
327  Analysis::AddAnalysisTime(DUR(analysis_time));
328  CLOCK(convert_time);
329  FaultTreeAnalysis::Convert(algorithm_->products(), graph_.get());
330  LOG(DEBUG2) << "Converted indices to pointers in " << DUR(convert_time);
331 }
