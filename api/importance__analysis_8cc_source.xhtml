36 void ImportanceAnalysis::Analyze() noexcept {
37  CLOCK(imp_time);
38  LOG(DEBUG3) << "Calculating importance factors...";
39  std::vector<std::pair<int, const mef::BasicEvent*>> target_events =
40  this->GatherImportantEvents();
41  double p_total = this->p_total();
42  for (const auto& event : target_events) {
43  double p_var = event.second->p();
44  ImportanceFactors imp;
45  imp.mif = this->CalculateMif(event.first);
46  imp.cif = p_var * imp.mif / p_total;
47  imp.raw = 1 + (1 - p_var) * imp.mif / p_total;
48  imp.dif = p_var * imp.raw;
49  imp.rrw = p_total / (p_total - p_var * imp.mif);
50  importance_.emplace(event.second->id(), imp);
51  important_events_.emplace_back(event.second, imp);
52  }
53  LOG(DEBUG3) << "Calculated importance factors in " << DUR(imp_time);
54  Analysis::AddAnalysisTime(DUR(imp_time));
55 }

57 std::vector<std::pair<int, const mef::BasicEvent*>>
58 ImportanceAnalysis::GatherImportantEvents(
59  const BooleanGraph* graph,
60  const std::vector<std::vector<int>>& products) noexcept {
61  std::vector<std::pair<int, const mef::BasicEvent*>> important_events;
62  std::unordered_set<int> unique_indices;
63  for (const auto& product : products) {
64  for (int index : product) {
65  int pos_index = std::abs(index);
66  if (unique_indices.insert(pos_index).second == false)
67  continue;
68  important_events.emplace_back(pos_index,
69  graph->GetBasicEvent(pos_index));
70  }
71  }
72  return important_events;
73 }

76  Bdd::VertexPtr root = bdd_graph_->root().vertex;

87 double ImportanceAnalyzer<Bdd>::CalculateMif(const Bdd::VertexPtr& vertex,
88  int order, bool mark) noexcept {
89  if (vertex->terminal())
90  return 0;
91  ItePtr ite = Ite::Ptr(vertex);
92  if (ite->mark() == mark)
93  return ite->factor();
94  ite->mark(mark);
95  if (ite->order() > order) {
96  if (!ite->module()) {
97  ite->factor(0);
98  } else { /// @todo Detect if the variable is in the module.
99  // The assumption is
100  // that the order of a module is always larger
101  // than the order of its variables.
102  double high = RetrieveProbability(ite->high());
103  double low = RetrieveProbability(ite->low());
104  if (ite->complement_edge())
105  low = 1 - low;
106  const Bdd::Function& res =
107  bdd_graph_->modules().find(ite->index())->second;
108  double mif = CalculateMif(res.vertex, order, mark);
109  if (res.complement)
110  mif = -mif;
111  ite->factor((high - low) * mif);
112  }
113  } else if (ite->order() == order) {
114  assert(!ite->module() && "A variable can't be a module.");
115  double high = RetrieveProbability(ite->high());
116  double low = RetrieveProbability(ite->low());
117  if (ite->complement_edge())
118  low = 1 - low;
119  ite->factor(high - low);
120  } else {
121  assert(ite->order() < order);
122  double p_var = 0;
123  if (ite->module()) {
124  const Bdd::Function& res =
125  bdd_graph_->modules().find(ite->index())->second;
126  p_var = RetrieveProbability(res.vertex);
127  if (res.complement)
128  p_var = 1 - p_var;
129  } else {
130  p_var = prob_analyzer()->p_vars()[ite->index()];
131  }
132  double high = CalculateMif(ite->high(), order, mark);
133  double low = CalculateMif(ite->low(), order, mark);
134  if (ite->complement_edge())
135  low = -low;
136  ite->factor(p_var * high + (1 - p_var) * low);
137  }
138  return ite->factor();
139 }
