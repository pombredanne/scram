49 double CutSetProbabilityCalculator::Calculate(
50  const CutSet& cut_set,
51  const std::vector<double>& p_vars) noexcept {
52  if (cut_set.empty())
53  return 0;
54  double p_sub_set = 1; // 1 is for multiplication.
55  for (int member : cut_set) {
56  assert(member > 0 && "Complements in a cut set.");
57  p_sub_set *= p_vars[member];
58  }
59  return p_sub_set;

62 double RareEventCalculator::Calculate(
63  const std::vector<CutSet>& cut_sets,
64  const std::vector<double>& p_vars) noexcept {
65  if (CutSetProbabilityCalculator::CheckUnity(cut_sets))
66  return 1;
67  double sum = 0;
68  for (const auto& cut_set : cut_sets) {
69  assert(!cut_set.empty() && "Detected an empty cut set.");
70  sum += CutSetProbabilityCalculator::Calculate(cut_set, p_vars);
71  }
72  return sum > 1 ? 1 : sum;
73 }

75 double McubCalculator::Calculate(
76  const std::vector<CutSet>& cut_sets,
77  const std::vector<double>& p_vars) noexcept {
78  if (CutSetProbabilityCalculator::CheckUnity(cut_sets))
79  return 1;
80  double m = 1;
81  for (const auto& cut_set : cut_sets) {
82  assert(!cut_set.empty() && "Detected an empty cut set.");
83  m *= 1 - CutSetProbabilityCalculator::Calculate(cut_set, p_vars);

85  return 1 - m;

109 double ProbabilityAnalyzer<Bdd>::CalculateTotalProbability(
110  const std::vector<double>& p_vars) noexcept {
111  CLOCK(calc_time); // BDD based calculation time.
112  LOG(DEBUG4) << "Calculating probability with BDD...";
113  current_mark_ = !current_mark_;
114  double prob =
115  CalculateProbability(bdd_graph_->root().vertex, current_mark_, p_vars);
116  if (bdd_graph_->root().complement)
117  prob = 1 - prob;
118  LOG(DEBUG4) << "Calculated probability " << prob << " in " << DUR(calc_time);
119  return prob;
120 }

147 double ProbabilityAnalyzer<Bdd>::CalculateProbability(
148  const Bdd::VertexPtr& vertex,
149  bool mark,
150  const std::vector<double>& p_vars) noexcept {
151  if (vertex->terminal())
152  return 1;
153  ItePtr ite = Ite::Ptr(vertex);
154  if (ite->mark() == mark)
155  return ite->p();
156  ite->mark(mark);
157  double p_var = 0;
158  if (ite->module()) {
159  const Bdd::Function& res = bdd_graph_->modules().find(ite->index())->second;
160  p_var = CalculateProbability(res.vertex, mark, p_vars);
161  if (res.complement)
162  p_var = 1 - p_var;
163  } else {
164  p_var = p_vars[ite->index()];
165  }
166  double high = CalculateProbability(ite->high(), mark, p_vars);
167  double low = CalculateProbability(ite->low(), mark, p_vars);
168  if (ite->complement_edge())
169  low = 1 - low;
170  ite->p(p_var * high + (1 - p_var) * low);
171  return ite->p();
172 }
