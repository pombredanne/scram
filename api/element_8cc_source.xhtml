63 const Attribute& Element::GetAttribute(const std::string& name) const {
64  auto it = boost::find_if(attributes_, [&name](const Attribute& attr) {
65  return attr.name == name;
66  });
67  if (it == attributes_.end())
68  throw LogicError("Element does not have attribute: " + name);
69 
70  return *it;
