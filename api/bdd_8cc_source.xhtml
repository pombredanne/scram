42 Bdd::Bdd(const BooleanGraph* fault_tree, const Settings& settings)
43  : kSettings_(settings),
44  coherent_(fault_tree->coherent()),
45  kOne_(new Terminal<Ite>(true)),
46  function_id_(2) {
47  CLOCK(init_time);
48  LOG(DEBUG3) << "Converting Boolean graph into BDD...";
49  if (fault_tree->root()->IsConstant()) {
50  // Constant case should only happen to the top gate.
51  if (fault_tree->root()->state() == kNullState) {
52  root_ = {true, kOne_};
53  } else {
54  root_ = {false, kOne_};
55  }
56  } else if (fault_tree->root()->type() == kNull) {
57  const GatePtr& top_gate = fault_tree->root();
58  assert(top_gate->args().size() == 1);
59  assert(top_gate->args<Gate>().empty());
60  int child = *top_gate->args().begin();
61  VariablePtr var = top_gate->args<Variable>().begin()->second;
62  root_ = {child < 0,
63  FindOrAddVertex(var->index(), kOne_, kOne_, true, var->order())};
64  } else {
65  std::unordered_map<int, std::pair<Function, int>> gates;
66  root_ = ConvertGraph(*fault_tree->root(), &gates);
67  root_.complement ^= fault_tree->complement();
68  }
69  ClearMarks(false);
70  TestStructure(root_.vertex);
71  LOG(DEBUG4) << "# of BDD vertices created: " << function_id_ - 1;
72  LOG(DEBUG4) << "# of entries in unique table: " << unique_table_.size();
73  LOG(DEBUG4) << "# of entries in AND table: " << and_table_.size();
74  LOG(DEBUG4) << "# of entries in OR table: " << or_table_.size();
75  ClearMarks(false);
76  LOG(DEBUG4) << "# of ITE in BDD: " << CountIteNodes(root_.vertex);
77  LOG(DEBUG3) << "Finished Boolean graph conversion in " << DUR(init_time);
78  ClearMarks(false);
79  // Clear tables if no more calculations are expected.
80  ClearTables();
81  if (coherent_) {
82  unique_table_.Release();
83  and_table_.reserve(0);
84  or_table_.reserve(0);
85  }
86 }

106 ItePtr Bdd::FindOrAddVertex(int index, const VertexPtr& high,
107  const VertexPtr& low, bool complement_edge,
108  int order) noexcept {
109  assert(index > 0 && "Only positive indices are expected.");
110  IteWeakPtr& in_table =
111  unique_table_.FindOrAdd(index, high->id(),
112  complement_edge ? -low->id() : low->id());
113  if (!in_table.expired())
114  return in_table.lock();
115  assert(order > 0 && "Improper order.");
116  ItePtr ite(new Ite(index, order, function_id_++, high, low));
117  ite->complement_edge(complement_edge);
118  in_table = ite;
119  return ite;

151 Bdd::Function Bdd::ConvertGraph(
152  const Gate& gate,
153  std::unordered_map<int, std::pair<Function, int>>* gates) noexcept {
154  assert(!gate.IsConstant() && "Unexpected constant gate!");
155  Function result; // For the NRVO, due to memoization.
156  // Memoization check.
157  if (auto it_entry = ext::find(*gates, gate.index())) {
158  std::pair<Function, int>& entry = it_entry->second;
159  result = entry.first;
160  assert(entry.second < gate.parents().size()); // Processed parents.
161  if (++entry.second == gate.parents().size())
162  gates->erase(it_entry);
163  return result;
164  }
165  std::vector<Function> args;
166  for (const Gate::Arg<Variable>& arg : gate.args<Variable>()) {
167  args.push_back(
168  {arg.first < 0, FindOrAddVertex(arg.second->index(), kOne_, kOne_, true,
169  arg.second->order())});
170  index_to_order_.emplace(arg.second->index(), arg.second->order());
171  }
172  for (const Gate::Arg<Gate>& arg : gate.args<Gate>()) {
173  Function res = ConvertGraph(*arg.second, gates);
174  if (arg.second->module()) {
175  args.push_back(
176  {arg.first < 0, FindOrAddVertex(*arg.second, kOne_, kOne_, true)});
177  } else {
178  bool complement = (arg.first < 0) ^ res.complement;
179  args.push_back({complement, res.vertex});
180  }
181  }
182  boost::sort(args, [](const Function& lhs, const Function& rhs) {
183  if (lhs.vertex->terminal())
184  return true;
185  if (rhs.vertex->terminal())
186  return false;
187  return Ite::Ptr(lhs.vertex)->order() > Ite::Ptr(rhs.vertex)->order();
188  });
189  auto it = args.cbegin();
190  for (result = *it++; it != args.cend(); ++it) {
191  result = Apply(gate.type(), result.vertex, it->vertex, result.complement,
192  it->complement);
193  }
194  ClearTables();
195  assert(result.vertex);
196  if (gate.module())
197  modules_.emplace(gate.index(), result);
198  if (gate.parents().size() > 1)
199  gates->insert({gate.index(), {result, 1}});
200  return result;
201 }

203 std::pair<int, int> Bdd::GetMinMaxId(const VertexPtr& arg_one,
204  const VertexPtr& arg_two,
205  bool complement_one,
206  bool complement_two) noexcept {
207  assert(!arg_one->terminal() && !arg_two->terminal());
208  assert(arg_one->id() && arg_two->id());
209  assert(arg_one->id() != arg_two->id());
210  int min_id = arg_one->id() * (complement_one ? -1 : 1);
211  int max_id = arg_two->id() * (complement_two ? -1 : 1);
212  if (arg_one->id() > arg_two->id())
213  std::swap(min_id, max_id);
214  return {min_id, max_id};
215 }

217 /// Specialization of Apply for AND operator with BDD vertices.
218 template <>
219 Bdd::Function Bdd::Apply<kAnd>(const VertexPtr& arg_one,
220  const VertexPtr& arg_two, bool complement_one,
221  bool complement_two) noexcept {
222  assert(arg_one->id() && arg_two->id()); // Both are reduced function graphs.
223  if (arg_one->terminal()) {
224  if (complement_one)
225  return {true, kOne_};
226  return {complement_two, arg_two};
227  }
228  if (arg_two->terminal()) {
229  if (complement_two)
230  return {true, kOne_};
231  return {complement_one, arg_one};
232  }
233  if (arg_one->id() == arg_two->id()) { // Reduction detection.
234  if (complement_one ^ complement_two)
235  return {true, kOne_};
236  return {complement_one, arg_one};
237  }
238  std::pair<int, int> min_max_id =
239  GetMinMaxId(arg_one, arg_two, complement_one, complement_two);
240  if (auto it = ext::find(and_table_, min_max_id))
241  return it->second;
242  Function result = Apply<kAnd>(Ite::Ptr(arg_one), Ite::Ptr(arg_two),
243  complement_one, complement_two);
244  and_table_.emplace(min_max_id, result);
245  return result;
246 }

249 template <>
250 Bdd::Function Bdd::Apply<kOr>(const VertexPtr& arg_one,
251  const VertexPtr& arg_two, bool complement_one,
252  bool complement_two) noexcept {
253  assert(arg_one->id() && arg_two->id()); // Both are reduced function graphs.
254  if (arg_one->terminal()) {
255  if (!complement_one)
256  return {false, kOne_};
257  return {complement_two, arg_two};
258  }
259  if (arg_two->terminal()) {
260  if (!complement_two)
261  return {false, kOne_};
262  return {complement_one, arg_one};
263  }
264  if (arg_one->id() == arg_two->id()) { // Reduction detection.
265  if (complement_one ^ complement_two)

271  if (auto it = ext::find(or_table_, min_max_id))

279 template <Operator Type>
280 Bdd::Function Bdd::Apply(ItePtr ite_one, ItePtr ite_two,
281  bool complement_one,
282  bool complement_two) noexcept {
283  if (ite_one->order() > ite_two->order()) {
284  ite_one.swap(ite_two);
285  std::swap(complement_one, complement_two);
286  }
287 
288  Function high;
289  Function low;
290  if (ite_one->order() == ite_two->order()) { // The same variable.
291  assert(ite_one->index() == ite_two->index());
292  high = Apply<Type>(ite_one->high(), ite_two->high(), complement_one,
293  complement_two);
294  low = Apply<Type>(ite_one->low(), ite_two->low(),
295  complement_one ^ ite_one->complement_edge(),
296  complement_two ^ ite_two->complement_edge());
297  } else {
298  assert(ite_one->order() < ite_two->order());
299  high =
300  Apply<Type>(ite_one->high(), ite_two, complement_one, complement_two);
301  low = Apply<Type>(ite_one->low(), ite_two,
302  complement_one ^ ite_one->complement_edge(),
303  complement_two);
304  }
305 
306  bool complement_edge = high.complement ^ low.complement;
307  if (complement_edge || (high.vertex->id() != low.vertex->id())) {
308  high.vertex =
309  FindOrAddVertex(ite_one, high.vertex, low.vertex, complement_edge);
310  }
311 
312  return high;
313 }

363 void Bdd::TestStructure(const VertexPtr& vertex) noexcept {
364  if (vertex->terminal())
365  return;
366  ItePtr ite = Ite::Ptr(vertex);
367  if (ite->mark())
368  return;
369  ite->mark(true);
370  assert(ite->index() && "Illegal index for a node.");
371  assert(ite->order() && "Improper order for nodes.");
372  assert(ite->high() && ite->low() && "Malformed node high/low pointers.");
373  assert(
374  !(!ite->complement_edge() && ite->high()->id() == ite->low()->id()) &&
375  "Reduction rule failure.");
376  assert(!(!ite->high()->terminal() &&
377  ite->order() >= Ite::Ptr(ite->high())->order()) &&
378  "Ordering of nodes failed.");
379  assert(!(!ite->low()->terminal() &&
380  ite->order() >= Ite::Ptr(ite->low())->order()) &&
381  "Ordering of nodes failed.");
382  if (ite->module()) {
383  const Function& res = modules_.find(ite->index())->second;
384  assert(!res.vertex->terminal() && "Terminal modules must be removed.");
385  TestStructure(res.vertex);
386  }
387  TestStructure(ite->high());
388  TestStructure(ite->low());
389 }
