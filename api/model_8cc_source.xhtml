94 std::shared_ptr<T> Model::GetEntity(const std::string& entity_reference,
95  const std::string& base_path,
96  const LookupTable<T>& container) {
97  assert(!entity_reference.empty());
98  if (!base_path.empty()) { // Check the local scope.
99  if (auto it = ext::find(container.entities_by_path,
100  base_path + "." + entity_reference))
101  return *it;
102  }
103 
104  auto at = [&entity_reference](const auto& reference_container) {
105  if (auto it = ext::find(reference_container, entity_reference))
106  return *it;
107  throw std::out_of_range("The event cannot be found.");
108  };
109 
110  if (entity_reference.find('.') == std::string::npos) // Public entity.
111  return at(container.entities_by_id);
112 
113  return at(container.entities_by_path); // Direct access.
114 }

125 void Model::BindEvent(const std::string& entity_reference,
126  const std::string& base_path, Formula* formula) {
127  assert(!entity_reference.empty());
128  if (!base_path.empty()) { // Check the local scope.
129  std::string full_path = base_path + "." + entity_reference;
130  BIND_EVENT(entities_by_path, full_path);
131  }
132 
133  if (entity_reference.find('.') == std::string::npos) { // Public entity.
134  BIND_EVENT(entities_by_id, entity_reference);
135  } else { // Direct access.
136  BIND_EVENT(entities_by_path, entity_reference);
137  }
138  throw std::out_of_range("The event cannot be bound.");
139 }
