32 /// Negative exponential distribution
33 /// with hourly failure rate and time.
34 class ExponentialExpression : public Expression {
35  public:
36  /// Constructor for exponential expression with two arguments.
37  ///
38  /// @param[in] lambda Hourly rate of failure.
39  /// @param[in] t Mission time in hours.
40  ExponentialExpression(const ExpressionPtr& lambda, const ExpressionPtr& t);
41 
42  /// @throws InvalidArgument The failure rate or time is negative.
43  void Validate() const override;
44 
45  double Mean() noexcept override {
46  return 1 - std::exp(-(lambda_.Mean() * time_.Mean()));
47  }
48 
49  double Max() noexcept override {
50  return 1 - std::exp(-(lambda_.Max() * time_.Max()));
51  }
52 
53  double Min() noexcept override {
54  return 1 - std::exp(-(lambda_.Min() * time_.Min()));
55  }
56 
57  private:
58  double GetSample() noexcept override {
59  return 1 - std::exp(-(lambda_.Sample() * time_.Sample()));
60  }
61 
62  Expression& lambda_; ///< Failure rate in hours.
63  Expression& time_; ///< Mission time in hours.
64 };
