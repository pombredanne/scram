28 Settings& Settings::algorithm(const std::string& algorithm) {
29  if (algorithm != "mocus" && algorithm != "bdd" && algorithm != "zbdd")
30  throw InvalidArgument(
31  "The qualitative analysis algorithm is not recognized.");
32 
33  algorithm_ = algorithm;
34  if (algorithm_ == "bdd") {
35  approximation("no");
36  } else {
37  if (approximation_ == "no")
38  approximation("rare-event");
39  if (prime_implicants_)
40  prime_implicants(false);
41  }
42  return *this;
43 }
