37 void Reporter::Report(const core::RiskAnalysis& risk_an, std::ostream& out) {
38  out << "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n";
39  XmlStreamElement report("report", out);
40  ReportInformation(risk_an, &report);
41 
42  CLOCK(report_time);
43  LOG(DEBUG1) << "Reporting analysis results...";
44  XmlStreamElement results = report.AddChild("results");
45  for (const auto& fta : risk_an.fault_tree_analyses()) {
46  const std::string& id = fta.first;
47  core::ProbabilityAnalysis* prob_analysis = nullptr; // Null if no analysis.
48  if (risk_an.settings().probability_analysis()) {
49  prob_analysis = risk_an.probability_analyses().at(id).get();
50  }
51  ReportResults(id, *fta.second, prob_analysis, &results);
52 
53  if (risk_an.settings().importance_analysis()) {
54  ReportResults(id, *risk_an.importance_analyses().at(id), &results);
55  }
56 
57  if (risk_an.settings().uncertainty_analysis()) {
58  ReportResults(id, *risk_an.uncertainty_analyses().at(id), &results);
59  }
60  }
61  LOG(DEBUG1) << "Finished reporting in " << DUR(report_time);
62 }

267 void Reporter::ReportResults(const std::string& ft_name,
268  const core::FaultTreeAnalysis& fta,
269  const core::ProbabilityAnalysis* prob_analysis,
270  XmlStreamElement* results) {
271  XmlStreamElement sum_of_products = results->AddChild("sum-of-products");
272  sum_of_products.SetAttribute("name", ft_name)
273  .SetAttribute("basic-events", fta.product_events().size())
274  .SetAttribute("products", fta.products().size());
275 
276  std::string warning = fta.warnings();
277  if (prob_analysis) {
278  sum_of_products.SetAttribute("probability", prob_analysis->p_total());
279  warning += prob_analysis->warnings();
280  }
281  if (!warning.empty()) {
282  sum_of_products.AddChild("warning").AddText(warning);
283  }
284 
285  CLOCK(cs_time);
286  LOG(DEBUG2) << "Reporting products for " << ft_name << "...";
287  std::vector<double> probs; // Product probabilities.
288  double sum = 0; // Sum of probabilities for contribution calculations.
289  if (prob_analysis) {
290  for (const core::Product& product_set : fta.products()) {
291  double prob = CalculateProbability(product_set);
292  sum += prob;
293  probs.push_back(prob);
294  }
295  } // Ugliness because FTA and Probability analyses are not integrated.
296  for (int i = 0; i < fta.products().size(); ++i) {
297  const core::Product& product_set = fta.products()[i];
298  XmlStreamElement product = sum_of_products.AddChild("product");
299  product.SetAttribute("order", GetOrder(product_set));
300  if (prob_analysis) {
301  double prob = probs[i];
302  product.SetAttribute("probability", prob);
303  product.SetAttribute("contribution", prob / sum);
304  }
305  for (const core::Literal& literal : product_set) {
306  ReportLiteral(literal, &product);
307  }
308  }
309  LOG(DEBUG2) << "Finished reporting products in " << DUR(cs_time);
310 }

338 void Reporter::ReportResults(const std::string& ft_name,
339  const core::UncertaintyAnalysis& uncert_analysis,
340  XmlStreamElement* results) {
341  XmlStreamElement measure = results->AddChild("measure");
342  measure.SetAttribute("name", ft_name);
343  if (!uncert_analysis.warnings().empty()) {
344  measure.AddChild("warning").AddText(uncert_analysis.warnings());
345  }
346  measure.AddChild("mean").SetAttribute("value", uncert_analysis.mean());
347  measure.AddChild("standard-deviation")
348  .SetAttribute("value", uncert_analysis.sigma());
349 
350  measure.AddChild("confidence-range")
351  .SetAttribute("percentage", "95")
352  .SetAttribute("lower-bound", uncert_analysis.confidence_interval().first)
353  .SetAttribute("upper-bound",
354  uncert_analysis.confidence_interval().second);
355 
356  measure.AddChild("error-factor")
357  .SetAttribute("percentage", "95")
358  .SetAttribute("value", uncert_analysis.error_factor());
359  {
360  XmlStreamElement quantiles = measure.AddChild("quantiles");
361  int num_quantiles = uncert_analysis.quantiles().size();
362  quantiles.SetAttribute("number", num_quantiles);
363  double prev_bound = 0;
364  double delta = 1.0 / num_quantiles;
365  for (int i = 0; i < num_quantiles; ++i) {
366  double upper = uncert_analysis.quantiles()[i];
367  double value = delta * (i + 1);
368  quantiles.AddChild("quantile")
369  .SetAttribute("number", i + 1)
370  .SetAttribute("value", value)
371  .SetAttribute("lower-bound", prev_bound)
372  .SetAttribute("upper-bound", upper);
373  prev_bound = upper;
374  }
375  }
376  {
377  XmlStreamElement hist = measure.AddChild("histogram");
378  int num_bins = uncert_analysis.distribution().size() - 1;
379  hist.SetAttribute("number", num_bins);
380  for (int i = 0; i < num_bins; ++i) {
381  double lower = uncert_analysis.distribution()[i].first;
382  double upper = uncert_analysis.distribution()[i + 1].first;
383  double value = uncert_analysis.distribution()[i].second;
384  hist.AddChild("bin")
385  .SetAttribute("number", i + 1)
386  .SetAttribute("value", value)
387  .SetAttribute("lower-bound", lower)
388  .SetAttribute("upper-bound", upper);
389  }
390  }
391 }

404 template <class T>
405 void Reporter::ReportBasicEvent(const mef::BasicEvent& basic_event,
406  XmlStreamElement* parent, const T& add_data) {
407  const auto* ccf_event = dynamic_cast<const mef::CcfEvent*>(&basic_event);
408  if (!ccf_event) {
409  XmlStreamElement element = parent->AddChild("basic-event");
410  element.SetAttribute("name", basic_event.id());
411  add_data(&element);
412  } else {
413  const mef::CcfGroup& ccf_group = ccf_event->ccf_group();
414  XmlStreamElement element = parent->AddChild("ccf-event");
415  element.SetAttribute("ccf-group", ccf_group.id())
416  .SetAttribute("order", ccf_event->members().size())
417  .SetAttribute("group-size", ccf_group.members().size());
418  add_data(&element);
419  for (const mef::Gate* member : ccf_event->members()) {
420  element.AddChild("basic-event").SetAttribute("name", member->name());
421  }
422  }
