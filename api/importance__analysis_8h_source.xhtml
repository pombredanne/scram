179 template <class Calculator>
180 double ImportanceAnalyzer<Calculator>::CalculateMif(int index) noexcept {
181  using Base = ImportanceAnalyzerBase<Calculator>;
182 
183  double p_store = p_vars_[index]; // Save the original value for restoring.
184 
185  // Calculate P(top/event)
186  p_vars_[index] = 1;
187  double p_e = Base::prob_analyzer()->CalculateTotalProbability(p_vars_);
188  assert(p_e >= 0 && p_e <= 1);
189 
190  // Calculate P(top/Not event)
191  p_vars_[index] = 0;
192  double p_not_e = Base::prob_analyzer()->CalculateTotalProbability(p_vars_);
193  assert(p_not_e >= 0 && p_not_e <= 1);
194 
195  // Restore the probability for next importance calculation.
196  p_vars_[index] = p_store;
197 
198  return p_e - p_not_e;
199 }
